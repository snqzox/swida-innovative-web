all programmatic.

// quote → book → callback. Done.
import { Swida } from "@swida/instacarrier";

const swida = new Swida(process.env.SWIDA_KEY);

const quote = await swida.quote({
  from: "Bratislava, SK",
  to:   "Lyon, FR",
  vehicle: "van_3_5t",
  weightKg: 1200
});

await swida.book(quote.id);
